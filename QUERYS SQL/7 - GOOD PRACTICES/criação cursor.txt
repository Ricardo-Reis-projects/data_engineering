DECLARE @VARIAVEL NVARCHAR(200)
DECLARE @SQL_COMMAND NVARCHAR(MAX)

DECLARE DB_CURSOR CURSOR FOR
-- LISTA DE CONTROLE 
select [name] from SYS.objects

-- ABRE O CURSOR
OPEN DB_CURSOR

-- INICIA O CURSOR
FETCH NEXT FROM DB_CURSOR INTO @VARIAVEL

-- ZERA
WHILE @@FETCH_STATUS = 0
-- BLOCO DE INSTRUCOES
	BEGIN
	PRINT @VARIAVEL 

-- SETA NOVO ITEM DA LISTA	
	    FETCH NEXT FROM DB_CURSOR INTO @VARIAVEL
	END
-- FECHA O CURSOR
CLOSE DB_CURSOR

-- LIBERA A MEMORIA
DEALLOCATE DB_CURSOR
